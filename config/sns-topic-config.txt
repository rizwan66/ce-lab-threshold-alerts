Creation Commands
-----------------
# Create topic
TOPIC_ARN=$(aws sns create-topic \
  --name CloudWatchAlerts \
  --tags Key=Environment,Value=Production \
  --query 'TopicArn' \
  --output text)

# Subscribe email
aws sns subscribe \
  --topic-arn $TOPIC_ARN \
  --protocol email \
  --notification-endpoint rizwan.nasir@hotmail.com

# Verify subscription
aws sns list-subscriptions-by-topic --topic-arn $TOPIC_ARN
